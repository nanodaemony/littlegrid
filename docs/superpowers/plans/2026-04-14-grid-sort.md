```dart
  /// 获取排序后的工具
  List<ToolConfig> getSortedTools() {
    final sorted = List<ToolConfig>.from(_toolConfigs);

    // 按置顶和功能名称排序
    sorted.sort((a, b) {
      // 置顶优先
      if (a.isPinned && !b.isPinned) return -1;
      if (!a.isPinned && b.isPinned) return 1;

      // 按功能名称排序（字母/拼音顺序）
      return a.name.compareTo(b.name);
    });

    return sorted;
  }
```
